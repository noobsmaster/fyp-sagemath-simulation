m = row size of divided matrix
function FormPartialUpperTriangularMatrix(partialMat)
	resultMat = null matrix
	for i = 0 to m-1 do
		Search ith pivoting row from partialMat as ith pivotRow, delete row from partialMat
		if pivotRow not found :
			continue for loop(skip rest of this loop iteration)
		for row in partialMat do
			if ith entry of row is non-zero then
				row <- row XOR pivotRow
			end if
		end for
	end for
	if partialMat not empty:
		insert remaining rows of partialMat in to resultMat which pivotRow not found
	return resultMat
end function

function MergingPartialUpperTriangularMatrix(partialMatTop,partialMatBtm)
	for i = 0 to m-1 do
		if ith entry of ith row of partialMatTop == 0	
			assert ith entry of ith row of partialMatBtm == 1
			exchange ith row of partialMatTop with ith row of partialMatBtm
		end if
		for j = 0 to m-1 do
			if ith entry of jth row of partialMatBtm is non-zero then
				row <- row XOR ith row of partialMatTop
			end if
		end for
	end for
	return(partialMatTop,partialMatBtm)
end function
